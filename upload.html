local module = {}

local RunService = game:GetService("RunService")
local HttpService = game:GetService("HttpService")

module.GridSize = 4
module.AllowedInstances = {
  "Part",
  "MeshPart",
  "Model"
}

function module:SnapToGrid(position)
  local snapped = Vector3.new(
    math.floor(position.X / self.GridSize + 0.5) * self.GridSize,
    math.floor(position.Y / self.GridSize + 0.5) * self.GridSize,
    math.floor(position.Z / self.GridSize + 0.5) * self.GridSize
  )
  return snapped
end

return module
